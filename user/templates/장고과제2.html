from pecab import PeCab #한국어 형태소 분석기
from tqdm import tqdm
tqdm.pandas()
pecab = PeCab()
def pos_text(review_text): # 'pos_text' 함수: 리뷰 텍스트에서 특정 품사를 제거한 후, 남은 텍스트를 반환
    join_text = [] # 가공된 텍스트를 저장할 리스트
    for text, pos in pecab.pos(review_text, drop_space=False):  # PeCab의 'pos' 메서드를 사용해 텍스트를 형태소 분석
        if pos not in ["JKS", "JX", "SF"]: #JKS: 주격 조사, JX: 보조사, SF: 마침표)를 제외한 나머지 형태소를 결합
            join_text.append(text)
    return("".join(join_text)) # 가공된 텍스트를 하나의 문자열로 반환
df["content_pecab"] = df["content_remove_html"].progress_map(pos_text) # 'content_remove_html' 컬럼의 각 값을 'pos_text' 함수로 처리하여 결과를 'content_pecab' 컬럼에 저장

import matplotlib.pyplot as plt

# LDA 모델의 주제별 상위 단어 시각화를 위한 함수
def plot_top_words(model, feature_names, n_top_words, title): # plot_top_words 함수 정의
    fig, axes = plt.subplots(1, 5, figsize=(30, 15), sharex=True) # 1x5 서브플롯을 생성하고, 전체 그래프의 크기를 (30, 15)로 설정
    axes = axes.flatten() # 1차원 배열로 변환
    for topic_idx, topic in enumerate(model.components_): # enumerate 함수를 사용해 주제별로 반복
        top_features_ind = topic.argsort()[:-n_top_words - 1:-1] # 주제별로 상위 n_top_words개의 단어를 추출
        top_features = [feature_names[i] for i in top_features_ind] # 상위 단어의 인덱스를 통해 실제 단어를 찾아서 리스트로 저장
        weights = topic[top_features_ind] # 상위 단어의 인덱스에 해당하는 가중치(확률값)를 추출
        ax = axes[topic_idx] # 주제별로 서브플롯을 선택
        ax.barh(top_features, weights, height=0.7) # 가로 막대 그래프를 생성
        for i in "top right".split(): # 서브플롯의 상단과 우측의 테두리를 제거
            ax.spines[i].set_visible(False) 
        ax.set_title(f'Topic {topic_idx + 1}', fontdict={'fontsize': 30}) # 서브플롯의 제목을 설정
        ax.invert_yaxis() # y축을 내림차순으로 정렬
        ax.tick_params(axis='both', which='major', labelsize=20) # x축과 y축의 눈금 레이블 크기를 설정
    plt.suptitle(title, fontsize=30) # 전체 그래프의 제목을 설정
    plt.subplots_adjust(top=0.90, bottom=0.05, wspace=0.40, hspace=0.1) # 전체 그래프의 여백을 설정
    plt.show() # 그래프를 출력

# n_top_words 변수를 정의하고 함수 호출
n_top_words = 20 # 상위 20개의 단어를 추출
